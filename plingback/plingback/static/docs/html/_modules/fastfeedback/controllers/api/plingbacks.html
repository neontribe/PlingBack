class FeedbackAttribute(object):
    """Base Class for feedback attributes"""
    def __init__(self, feedback_id, data):
        self.feedback_id = feedback_id
        self.feedback_uri = make_feedback_uri(feedback_id)
        self.data = data
        
    def make_triples(self):
        """ Produce a list of rdflib triples ready to add to a graph"""
        raise NotImplementedError

    def removal_query(self):
        """ Produce a sparql query to construct a set of triple which will be 
        removed in order to update this attribute's data"""
        raise NotImplementedError
    
    
class AttendanceAttribute(FeedbackAttribute):
    """ Helper class for attendance attributes """
    
    def make_triples(self):
        value = self.data.getone('attendance_value')
        if value is not None:
            return [(self.feedback_uri, PBO['attendance'], PBO[value])]
        else:
            return []
        
    def removal_query(self):
        query = "PREFIX pbo: <%s> "
        query += "CONSTRUCT { <%s> pbo:attendance ?o } WHERE { <%s> pbo:attendance ?o}"
        query = query % (str(PBO), self.feedback_uri, self.feedback_uri)
        return query
    
class RatingAttribute(FeedbackAttribute):
    """Helper class for rating attributes"""
    
    def make_triples(self):
        value = self.data.getone('rating_value')
        try:
            rating = '.' in value and float(value) or int(value)
        except ValueError:
            abort(400, 
                      'Value "%s" cannot be converted to a numeric type' % (value))
        return [(self.feedback_uri, REV['rating'], Literal(rating))]
        
    def removal_query(self):
        query = "PREFIX rev: <%s> "
        query += "CONSTRUCT { <%s> rev:rating ?o } WHERE { <%s> rev:rating ?o}"
        query = query % (str(REV), self.feedback_uri, self.feedback_uri)
        return query

class ReviewerAttribute(FeedbackAttribute):
    
    def make_triples(self):

        triples = []
        reviewer = BNode()
        triples.append((self.feedback_uri, REV['reviewer'], reviewer))
        triples.append((reviewer, RDFS['type'], FOAF['Person']))
        # See if we've been provided an email address
        try:
            email = self.data.getone('reviewer_email')
            if email:
                triples.append((reviewer, FOAF['mbox'], Literal(email)))
        except KeyError:
            pass
        # See if we've been provided a phone number
        try:
            phone = self.data.getone('reviewer_phone')
            if phone:
                triples.append((reviewer, FOAF['phone'], Literal('tel:' + phone)))
        except KeyError:
            pass
        # See if we've been provided a birthday
        try:
            birthday = self.data.getone('reviewer_birthday')
            if birthday:
                triples.append((reviewer, FOAF['dateOfBirth'], Literal(birthday)))
        except KeyError:
            pass
        # See if we've been provided a third party id
        try:
            tpid = self.data.getone('reviewer_id')
            tps = self.data.getone('reviewer_id_source')
            if tpid or tps:
                acc = BNode()
                triples.append((reviewer, FOAF['holdsAccount'], acc))
                triples.append((acc, RDFS['type'], FOAF['OnlineAccount']))
                if tps:
                    triples.append((acc, FOAF['accountServiceHomepage'], URIRef(tps)))
                if tpid:
                    triples.append((acc, FOAF['accountName'], Literal(tpid)))
        except KeyError:
            pass
        
        return triples

    def removal_query(self):
        query = """
            PREFIX rev: <%s> 
            PREFIX rdfs: <%s> 
            PREFIX foaf: <%s> 
            CONSTRUCT { <%s> rev:reviewer ?rbn .
                                     ?rbn rdfs:type foaf:Person .
                                     ?rbn foaf:mbox ?email .
                                     ?rbn foaf:dateOfBirth ?dob.
                                     ?rbn foaf:holdsAccount ?acc.
                                     ?acc rdfs:type foaf:OnlineAccount.
                                     ?acc foaf:accountServiceHomepage ?homepage.
                                     ?acc foaf:accountName ?accid} 
            WHERE { <%s> rev:reviewer ?rbn .
                            OPTIONAL { 
                                               ?rbn foaf:mbox ?email .
                                               ?rbn foaf:dateOfBirth ?dob.
                                               ?rbn foaf:holdsAccount ?acc.
                                               ?acc foaf:accountServiceHomepage ?homepage.
                                               ?acc foaf:accountName ?accid
                              }
            }"""
        query = query % (str(REV), str(RDFS), str(FOAF), self.feedback_uri, self.feedback_uri)
        return query
               

def attribute_factory(name, feedback_id, data):
    """Returns an instance of an appropriate class to act as helper"""
    dispatcher = { 'comment' : CommentAttribute,
                             'rating' : RatingAttribute,
                             'reviewer':ReviewerAttribute,
                             'attendance':AttendanceAttribute}
    return dispatcher[name](feedback_id, data)

    def _get_unique_feedback_id(self):
        """ Get a unique id for a piece of feedback by using the Primary Key Pattern
        
        See http://n2.talis.com/wiki/Primary_Key_Pattern for an overview
        """
        store = Talis("plings-dev2", username="plings", password="ck6sjkfp")
        candidates = []
        attempts = 0
        feedback_id = None
        
        def fetch_candidates():
            """Get a list of potential ids"""
            candidates_query = """
              PREFIX pb: <http://plingback.plings.net/ontologies/plingback#>
              SELECT ?s  ?o WHERE { ?s pb:idCandidate ?o } LIMIT %s
            """
            candidates_query = candidates_query % (ID_CANDIDATE_LIMIT)
            return [x for x in store.query(candidates_query)]
            
        def take_id(candidate):
            """Push a changeset removing our chosen id and adding a replacement
            
            If this succeeds we can go ahead and use the id"""
            store.remove((candidate[0], PBO['idCandidate'], candidate[1]))
            store.add((BNode(), PBO['idCandidate'], Literal(candidate[1].toPython() + ID_POOL_SIZE)))
            store.sync()
            return candidate[1].toPython()
            
        while feedback_id == None and attempts < MAX_ID_ATTEMPTS:
            if not candidates:
                candidates = fetch_candidates()
                
            candidate = random.choice(candidates)
            candidates.remove(candidate)
            
            try:
                feedback_id = take_id(candidate)
            except:
                pass
            
            attempts = attempts + 1

        if attempts >= MAX_ID_ATTEMPTS:
            log.critical("max attempts exceeded while trying to grab a unique feedback id")
            
        return feedback_id
        
    def _init_feedback_node(self, graph, request, feedback_id=None,):
        """ Add triples for our feedback node to the supplied graph/store object
        
        If no feedback_id is passed the request will be inspected for one
        
        The data will be extracted from the request which should contain the 
        following keys:
        
        pling_id : the id of the activity under review
        plingback_type : the application submitting the feedback
        plingback_version : the version of the application (OPTIONAL)
        feedback_id : the unique id to identify this feedback node (OPTIONAL)
        
        """
        # Validation!
        if not feedback_id:
            feedback_id = request.params.get('feedback_id', None)
            if not feedback_id:
                raise MissingFeedbackIDError("No Feedback Id was specified and" + \
                    "none was found in the request parameters")
        feedback_uri = make_feedback_uri(feedback_id)
        
                
        # Validation!
        if not request.params.get('pling_id'):
            abort(400, 'Please supply a pling_id')
        pling_id = request.params.get('pling_id')
        pling = URIRef('http://plings.net/a/%s' % (pling_id))

       
        plingback_type = request.params.get('plingback_type',
                                        'http://plingback.plings.net/applications/fastfeedback' )
        plingback_type_uri = URIRef(plingback_type)
        
        graph.add((feedback_uri, RDFS['type'], REV['Review']))
        graph.add((feedback_uri, PBO['isAbout'], pling))
        graph.add((feedback_uri, DC['date'], Literal(datetime.datetime.now())))
        
        graph.add((feedback_uri, PBO['plingBackType'], plingback_type_uri))
        if request.params.get('plingback_version'):
            graph.add((feedback_uri, 
                               PBO['plingBackVersion'], 
                               Literal(request.params.get('plingback_version'))))
            
        return graph
        


[docs]    def create(self, feedback_id=None):
        """POST /api/plingbacks: Create a new feedback node generating an id
            PUT /api/plingbacks/id: Create a new feedback node using id provided"""
       
        #Arbitrate between POST and PUT requests
        if request.method.lower() == 'post':
            feedback_id = self._get_unique_feedback_id()
            if not feedback_id:
                abort(500, 'Failed to generate a unique feedback_id')
        elif request.method.lower() == 'put':        
            if not feedback_id:
                if request.params.get('feedback_id'):
                    abort(400, 'the feedback_id should form part of the uri')
                else:
                    abort(400, 'Missing feedback_id')
        
        graph = configure_store()

        #Build the feedback node into the local graph
        self._init_feedback_node(graph, request, feedback_id)
        
        # Add any feedbackdata that we can find in the request
        self._add_feedback_attributes(graph, request, feedback_id)
        
        # Commit the changes to the store
        graph.sync()
        
        # We've done a create so add an appropriate Location header
        # and set the response code appropriately
        feedback_uri = make_feedback_uri(feedback_id)
        response.location = feedback_uri
        response.status = 201
        
        # Build a return json
        output = {'feedback_id': feedback_id,
                         'feedback_uri': feedback_uri}
        response.content_type = 'application/json'
        return json.dumps(output)        
            

    def _add_feedback_attributes(self, graph, request, feedback_id, attribute_name=None):   
        """ 
        """
        attributes = []
        if attribute_name:
            attributes = [attribute_name]
        else:
            attributes = request.params.getall('feedback_attribute')
            if not attributes:
                # An oddity in WebOb means that if multiple values were submitted for
                # a key via a javascript list we end up with odd key naming
                attributes = request.params.getall('feedback_attribute[]')
        
        for attribute in attributes:
            try:
                helper = attribute_factory(attribute, feedback_id, request.params)
                for triple in helper.make_triples():
                    graph.add(triple)
            except KeyError:
                abort(400, 'Unknown feedback attribute supplied')
            
        return graph
    
    def _remove_old_feedback_attribute(self, store_graph, feedback_id, attribute_name):
        """ Queries the store for the triple representing the attribute and then 
        issues removal requests against the graph"""
        local_graph = Graph()
        sparqurl = store_graph.urlbase + '/services/sparql?query='
        
        try:
            helper = attribute_factory(attribute_name, feedback_id, request.params)
            query = helper.removal_query()
            if query:
                local_graph.parse(sparqurl + urllib.quote_plus(query, '<>{}'))
                for triple in local_graph:
                    store_graph.remove(triple)
        except KeyError:
            abort(400, 'Unknown feedback attribute supplied')
                   
        return store_graph
            
            
            

[docs]    def add_attribute(self, feedback_id, attribute, overwrite=False):
        """ Record a feedback attribute for the given feedback node.
        
        @id : the unique id of the feedback node which the comment will be applied to
        @attribute: the name of the feedback attribute
        @update : overwrite the existing attribute for this node with the new one
                        (an HTTP PUT will trigger this)
        """
        graph = configure_store()
        
        self._add_feedback_attributes(graph, request, feedback_id, attribute)
        
        if overwrite:
            self._remove_old_feedback_attribute(graph, feedback_id, attribute)
        
        graph.sync()
        
        # Build a return json
        output = {'feedback_id':feedback_id,
                         'feedback_uri':make_feedback_uri(feedback_id)}
        response.content_type = 'application/json'
        return json.dumps(output)
